import java.util.*;

public class SortedSetExample {
    public static void main(String[] args) {
        // Create a SortedSet using TreeSet (which implements SortedSet)
        SortedSet<Integer> sortedSet = new TreeSet<>();

        // Adding elements to the SortedSet
        sortedSet.add(10);
        sortedSet.add(5);
        sortedSet.add(15);
        sortedSet.add(2);
        sortedSet.add(20);

        // Displaying the SortedSet
        System.out.println("SortedSet: " + sortedSet);

        // Accessing first and last elements
        System.out.println("First element: " + sortedSet.first());
        System.out.println("Last element: " + sortedSet.last());

        // Using headSet() to get elements less than 15
        SortedSet<Integer> headSet = sortedSet.headSet(15);
        System.out.println("HeadSet (less than 15): " + headSet);

        // Using tailSet() to get elements greater than or equal to 10
        SortedSet<Integer> tailSet = sortedSet.tailSet(10);
        System.out.println("TailSet (greater than or equal to 10): " + tailSet);

        // Using subSet() to get elements between 5 (inclusive) and 15 (exclusive)
        SortedSet<Integer> subSet = sortedSet.subSet(5, 15);
        System.out.println("SubSet (from 5 to 15): " + subSet);
    }
}
